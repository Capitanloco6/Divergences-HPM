#$STATE$ Joins $COUNTRY$
country_event = {
	id = 70000
	title = "EVTNAME70000"
	desc = "EVTDESC70000"
	picture = "Slaves"
	
	trigger = {
		country = {
			slavery = yes_slavery
			capital_scope = {
				NOT = { continent = europe }
			}
		}
	}
	
	is_triggered_only = yes
	allow_multiple_instances = yes
		
	option = {
		name = "EVTOPTA70000"
		is_slave = yes
		country = {
			any_state = {
				limit = {
					is_slave = no
				}
				any_pop = { consciousness = 2 }
			}
		}
		
	}
	
	option = {
		name = "EVTOPTB70000"
		is_slave = no
		country = {
			any_state = {
				limit = {
					is_slave = yes
				}
				any_pop = { consciousness = 2 }
			}		
		}
		
	}
}

# Nation that owns your debt goes bankrupt
country_event = {
	id = 70002
	title = "EVTNAME70002"
	desc = "EVTDESC70002"
	picture = "Bankruptcy"
	trigger = {
		NOT = { tag = FROM }
	}

	is_triggered_only = yes
	option = {
		name = "EVTOPTA70002"
		relation = { who = FROM value = -50 }
		random_owned = {
			limit = {
				owner = { is_greater_power = yes }
			}
			owner = {
				diplomatic_influence = { who = FROM value = 20 }
			}
		}
		random_country = {
			limit = {
				tag = FROM
				part_of_sphere = no
				is_vassal = no
				THIS = { is_greater_power = yes }
			}
			add_casus_belli = {
				target = THIS
				type = add_to_sphere
				months = 36
			}
		}
		random_country = {
			limit = {
				tag = FROM
				neighbour = THIS
				is_disarmed = no
				is_vassal = no
			}
			add_casus_belli = {
				target = THIS
				type = cut_down_to_size
				months = 12
			}
		}
	}
}
